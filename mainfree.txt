--  XỬ LÝ CÁC TOGGLE (Demo Coroutine)
-- ============================================================
local function runLoop(toggleName, interval, message)
    Options[toggleName]:OnChanged(function()
        local coro
        if Options[toggleName].Value then
            coro = coroutine.create(function()
                while Options[toggleName].Value do
                    print(message)
                    task.wait(interval or 3)
                end
            end)
            coroutine.resume(coro)
        else
            if coro then coroutine.close(coro); coro = nil end
        end
    end)
end
